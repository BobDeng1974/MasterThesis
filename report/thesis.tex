% ACKNOWLEDGEMENTS
%
% The acknowledgements environment puts a large, bold, centered 
% "Acknowledgements" label at the top of the page. The acknowledgements

\newenvironment{acknowledgements}

\vspace*{1.5cm}
{\Large \bfseries Acknowledgements}
\end{center}

{\end{quote}}

{\end{quote}}
